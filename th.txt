    // Script dactylo + lignes façon "no-cv-required"
    const steps = [
      { t: "> recherche de candidat...", d: 1000 },
      { t: "> filtrage des CVs...", d: 900 },
      { t: "> vérification de l’énergie... ", d: 700, add:"TROUVÉ"},
      { t: "> vérification de la motivation... ", d: 700, add:"TROUVÉ"},
      { t: "> vérification de la rapidité d’apprentissage... ", d: 800, add:"TROUVÉ"},
      { t: "> CANDIDAT TROUVÉ : YOUSSEF ERMADI", d: 1100, cls:"found"},
      { t: "> INITIALISATION DU PROTOCOLE DE CONTACT...", d: 1000 },
      { t: "> Bonjour. J’ai compris la mission. Construisons votre vision.", d: 1200 },
    ];

    const term = document.getElementById('terminal');

    function line(text, extra){
      const row = document.createElement('div');
      const prompt = document.createElement('span');
      prompt.className = 'prompt';
      prompt.textContent = '> ';
      const span = document.createElement('span');
      span.className = 'type';
      span.style.setProperty('--dur', Math.max(1.2, text.length/18) + 's');
      span.textContent = text.replace(/^\>\s?/, '');
      row.appendChild(prompt);
      row.appendChild(span);
      if(extra){ row.appendChild(extra); }
      term.appendChild(row);
    }

    (async function run(){
      term.innerHTML = '';
      for(const s of steps){
        let extra=null;
        if(s.add){
          extra = document.createElement('span');
          extra.textContent = s.add;
          extra.className = 'found';
          extra.style.marginLeft = '8px';
        }
        if(s.cls==='found'){
          const strong = document.createElement('span');
          strong.textContent = s.t.replace(/^\>\s?/, '');
          strong.className = 'found';
          line(' ', strong);
        }else{
          line(s.t);
        }
        await new Promise(r=>setTimeout(r, s.d));
      }
      // Ajout du lien à la fin (encore)
      const end = document.createElement('div');
      end.innerHTML = '<span class="prompt">&gt;</span> <a class="enter" href="#portfolio">Entrer dans le portfolio</a>';
      term.appendChild(end);
    })();

    // Scroll doux pour lien "Entrer"
    document.getElementById('enterLink').addEventListener('click', (e)=>{
      const hash = e.currentTarget.getAttribute('href');
      if(hash.startsWith('#')){
        e.preventDefault();
        document.querySelector(hash).scrollIntoView({behavior:'smooth'});
      }
    });
  </script>
